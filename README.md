# ISOGlyP

#Usage
isoglypCL.py [-h] [-f FASTA] [-a ACCESSIONS] [-p PARAMETERS] [-j JOBID]
                    [-c CUTOFF]

optional arguments:
  -h, --help            show this help message and exit
  -f FASTA, --fasta FASTA
  -a ACCESSIONS, --accessions ACCESSIONS
  -p PARAMETERS, --parameters PARAMETERS
  -j JOBID, --jobId JOBID
  -c CUTOFF, --cutoff CUTOFF
